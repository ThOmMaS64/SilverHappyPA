CREATE TABLE WORK_ADDRESS(
   ID_WORK_ADDRESS INTEGER PRIMARY KEY AUTO_INCREMENT,
   city VARCHAR(50) ,
   street VARCHAR(100) ,
   nb_street INT,
   postal_code CHAR(5) ,
   service_offered VARCHAR(150),
   silver_happy_local BOOLEAN DEFAULT 0,
   nb_boxes INT,
   nb_conference_rooms INT,
   nb_workshops INT
);

CREATE TABLE USER_(
   ID_USER INT AUTO_INCREMENT,
   username VARCHAR(50) UNIQUE,
   password VARCHAR(255) ,
   email VARCHAR(100) UNIQUE,
   name VARCHAR(40) ,
   surname VARCHAR(40) ,
   city VARCHAR(50) ,
   street VARCHAR(100) ,
   nb_street INT,
   postal_code CHAR(5) ,
   description VARCHAR(255) NOT NULL DEFAULT '',
   keyWord1 VARCHAR(50) NOT NULL DEFAULT '',
   keyWord2 VARCHAR(50) NOT NULL DEFAULT '',
   keyWord3 VARCHAR(50) NOT NULL DEFAULT '',
   status INT,
   date_inscription DATE,
   last_connection DATETIME DEFAULT '2000-01-01 00:00:00',
   darkMode INT DEFAULT 0,
   levelFont INT DEFAULT 0,
   fontChange INT DEFAULT 0,
   cursorType INT DEFAULT 0,
   language VARCHAR(5) DEFAULT 'fr',
   profilePicture VARCHAR(255) DEFAULT '',
   connected INT DEFAULT 0,
   banned BOOLEAN DEFAULT 0,
   PRIMARY KEY(ID_USER)
);

CREATE TABLE SUBSCRIPTION(
   ID_SUBSCRIPTION INT AUTO_INCREMENT,
   type VARCHAR(100),
   price DECIMAL(4,2)
   PRIMARY KEY(ID_SUBSCRIPTION)
);

CREATE TABLE SERVICE_PROVIDER(
   ID_SERVICE_PROVIDER INT AUTO_INCREMENT,
   profession VARCHAR(100) ,
   ID_USER INT NOT NULL,
   PRIMARY KEY(ID_SERVICE_PROVIDER),
   UNIQUE(ID_USER),
   FOREIGN KEY(ID_USER) REFERENCES USER_(ID_USER)
);

CREATE TABLE TOKEN(
   ID_TOKEN INT AUTO_INCREMENT,
   token VARCHAR(255) ,
   token_date DATETIME,
   ID_USER INT NOT NULL,
   PRIMARY KEY(ID_TOKEN),
   FOREIGN KEY(ID_USER) REFERENCES USER_(ID_USER)
);

CREATE TABLE ADVERTISEMENT(
   ID_ADVERTISEMENT INT AUTO_INCREMENT,
   ID_SERVICE_PROVIDER INT NOT NULL,
   ID_SUBSCRIPTION INT NOT NULL,
   type INT DEFAULT 0,
   stripe_subscription_id VARCHAR(255) DEFAULT NULL,
   title VARCHAR(100),
   description VARCHAR(255),
   PRIMARY KEY(ID_ADVERTISEMENT),
   FOREIGN KEY(ID_SERVICE_PROVIDER) REFERENCES SERVICE_PROVIDER(ID_SERVICE_PROVIDER),
   FOREIGN KEY(ID_SUBSCRIPTION) REFERENCES SUBSCRIPTION(ID_SUBSCRIPTION)
);

CREATE TABLE EVENT(
   ID_EVENT INT AUTO_INCREMENT,
   type VARCHAR(100) ,
   name VARCHAR(100),
   date_ DATETIME,
   description VARCHAR(200) ,
   ID_WORK_ADDRESS INT NOT NULL,
   PRIMARY KEY(ID_EVENT),
   FOREIGN KEY(ID_WORK_ADDRESS) REFERENCES WORK_ADDRESS(ID_WORK_ADDRESS)
);

CREATE TABLE CONSUMER(
   ID_CONSUMER INTEGER PRIMARY KEY AUTO_INCREMENT,
   birth_date DATE,
   tuto_seen BOOLEAN,
   ID_USER INT NOT NULL,
   ID_SUBSCRIPTION INT,
   stripe_subscription_id VARCHAR(255) DEFAULT NULL,
   UNIQUE(ID_USER),
   FOREIGN KEY(ID_USER) REFERENCES USER_(ID_USER),
   FOREIGN KEY(ID_SUBSCRIPTION) REFERENCES SUBSCRIPTION(ID_SUBSCRIPTION)
);

CREATE TABLE PRODUCT(
   ID_PRODUCT INT AUTO_INCREMENT,
   name VARCHAR(100),
   description VARCHAR(255),
   type VARCHAR(50) ,
   price DECIMAL(8,2)  ,
   PRIMARY KEY(ID_PRODUCT)
);

CREATE TABLE CONTRACT(
   ID_CONTRACT INT AUTO_INCREMENT,
   type VARCHAR(50) ,
   ID_CONSUMER INT,
   document_path VARCHAR(255),
   PRIMARY KEY(ID_CONTRACT),
   FOREIGN KEY(ID_CONSUMER) REFERENCES CONSUMER(ID_CONSUMER)
);

CREATE TABLE NOTIFICATION(
   ID_NOTIFICATION INT AUTO_INCREMENT,
   title VARCHAR(100) ,
   description VARCHAR(200) ,
   type VARCHAR(50) ,
   ID_CONSUMER INT NOT NULL,
   PRIMARY KEY(ID_NOTIFICATION),
   FOREIGN KEY(ID_CONSUMER) REFERENCES CONSUMER(ID_CONSUMER)
);

CREATE TABLE SERVICE(
   ID_SERVICE INT AUTO_INCREMENT,
   type VARCHAR(150) ,
   formation BOOLEAN,
   place VARCHAR(100) ,
   cost DECIMAL(8,2)  ,
   is_medical_confidential BOOLEAN DEFAULT 0,
   PRIMARY KEY(ID_SERVICE)
);

CREATE TABLE QUOTE(
   ID_QUOTE INT AUTO_INCREMENT,
   ID_SERVICE_PROVIDER INT NOT NULL,
   ID_CONSUMER INT NOT NULL,
   ID_SERVICE INT NOT NULL,
   amount DECIMAL(8,2)  ,
   date_ DATE,
   status BOOLEAN DEFAULT 0,
   PRIMARY KEY(ID_QUOTE),
   FOREIGN KEY (ID_SERVICE_PROVIDER) REFERENCES SERVICE_PROVIDER(ID_SERVICE_PROVIDER),
   FOREIGN KEY (ID_CONSUMER) REFERENCES CONSUMER(ID_CONSUMER),
   FOREIGN KEY (ID_SERVICE) REFERENCES SERVICE(ID_SERVICE)
);

CREATE TABLE ADVICE(
   ID_ADVICE INT AUTO_INCREMENT,
   title VARCHAR(100),
   theme VARCHAR(50) ,
   description TEXT,
   date_publication DATE,
   ID_SERVICE_PROVIDER INT NOT NULL,
   PRIMARY KEY(ID_ADVICE),
   FOREIGN KEY (ID_SERVICE_PROVIDER) REFERENCES SERVICE_PROVIDER(ID_SERVICE_PROVIDER)
);

CREATE TABLE MESSAGE(
   ID_MESSAGE INT AUTO_INCREMENT,
   content VARCHAR(200) ,
   date_ DATETIME,
   title VARCHAR(200) ,
   sender_id INT NOT NULL,
   receiver_id INT NOT NULL,
   PRIMARY KEY(ID_MESSAGE),
   FOREIGN KEY(sender_id) REFERENCES USER_(ID_USER),
   FOREIGN KEY(receiver_id) REFERENCES USER_(ID_USER)
);

CREATE TABLE PLANNING(
   ID_PLANNING INT PRIMARY KEY AUTO_INCREMENT,
   ID_USER INT NOT NULL,
   start_time DATETIME,
   end_time DATETIME,
   is_available BOOLEAN DEFAULT 1,
   FOREIGN KEY(ID_USER) REFERENCES USER_(ID_USER)
);

CREATE TABLE INVOICE(
   ID_INVOICE INT AUTO_INCREMENT,
   amount DECIMAL(8,2)  ,
   nb_services_provided INT,
   month_billed TINYINT,
   year_billed INT,
   pdf_path VARCHAR(255),
   ID_SERVICE_PROVIDER INT NOT NULL,
   PRIMARY KEY(ID_INVOICE),
   FOREIGN KEY(ID_SERVICE_PROVIDER) REFERENCES SERVICE_PROVIDER(ID_SERVICE_PROVIDER)
);

CREATE TABLE INTERVENTION(
   ID_INTERVENTION INT AUTO_INCREMENT,
   cost DECIMAL(8,2)  ,
   status TINYINT,
   start_date DATETIME,
   end_date DATETIME,
   commission DECIMAL (6,2),
   is_at_consumer_home BOOLEAN DEFAULT 0,
   transaction_id VARCHAR(255) DEFAULT NULL,
   ID_WORK_ADDRESS INT,
   ID_SERVICE INT NOT NULL,
   ID_INVOICE INT,
   PRIMARY KEY(ID_INTERVENTION),
   FOREIGN KEY(ID_WORK_ADDRESS) REFERENCES WORK_ADDRESS(ID_WORK_ADDRESS),
   FOREIGN KEY(ID_SERVICE) REFERENCES SERVICE(ID_SERVICE),
   FOREIGN KEY(ID_INVOICE) REFERENCES INVOICE(ID_INVOICE)
);

CREATE TABLE GRADE(
   ID_GRADE INT AUTO_INCREMENT,
   grade TINYINT,
   description VARCHAR(255) ,
   ID_INTERVENTION INT NOT NULL,
   ID_CONSUMER INT NOT NULL,
   PRIMARY KEY(ID_GRADE),
   FOREIGN KEY(ID_INTERVENTION) REFERENCES INTERVENTION(ID_INTERVENTION),
   FOREIGN KEY(ID_CONSUMER) REFERENCES CONSUMER(ID_CONSUMER)
);

CREATE TABLE PARTICIPATE(
   ID_EVENT INT,
   ID_CONSUMER INT,
   PRIMARY KEY(ID_EVENT, ID_CONSUMER),
   FOREIGN KEY(ID_EVENT) REFERENCES EVENT(ID_EVENT),
   FOREIGN KEY(ID_CONSUMER) REFERENCES CONSUMER(ID_CONSUMER)
);

CREATE TABLE CALL_(
   ID_CONSUMER INT,
   ID_INTERVENTION INT,
   PRIMARY KEY(ID_CONSUMER, ID_INTERVENTION),
   FOREIGN KEY(ID_CONSUMER) REFERENCES CONSUMER(ID_CONSUMER),
   FOREIGN KEY(ID_INTERVENTION) REFERENCES INTERVENTION(ID_INTERVENTION)
);

CREATE TABLE OFFER(
   ID_SERVICE_PROVIDER INT,
   ID_SERVICE INT,
   PRIMARY KEY(ID_SERVICE_PROVIDER, ID_SERVICE),
   FOREIGN KEY(ID_SERVICE_PROVIDER) REFERENCES SERVICE_PROVIDER(ID_SERVICE_PROVIDER),
   FOREIGN KEY(ID_SERVICE) REFERENCES SERVICE(ID_SERVICE)
);

CREATE TABLE PROVIDE(
   ID_INTERVENTION INT,
   ID_QUOTE INT,
   PRIMARY KEY(ID_INTERVENTION, ID_QUOTE),
   FOREIGN KEY(ID_INTERVENTION) REFERENCES INTERVENTION(ID_INTERVENTION),
   FOREIGN KEY(ID_QUOTE) REFERENCES QUOTE(ID_QUOTE)
);

CREATE TABLE DO(
   ID_SERVICE_PROVIDER INT,
   ID_INTERVENTION INT,
   PRIMARY KEY(ID_SERVICE_PROVIDER, ID_INTERVENTION),
   FOREIGN KEY(ID_SERVICE_PROVIDER) REFERENCES SERVICE_PROVIDER(ID_SERVICE_PROVIDER),
   FOREIGN KEY(ID_INTERVENTION) REFERENCES INTERVENTION(ID_INTERVENTION)
);

CREATE TABLE CONDUCT(
   ID_SERVICE_PROVIDER INT,
   ID_EVENT INT,
   PRIMARY KEY(ID_SERVICE_PROVIDER, ID_EVENT),
   FOREIGN KEY(ID_SERVICE_PROVIDER) REFERENCES SERVICE_PROVIDER(ID_SERVICE_PROVIDER),
   FOREIGN KEY(ID_EVENT) REFERENCES EVENT(ID_EVENT)
);

CREATE TABLE CAPTCHA(
   ID_CAPTCHA INT AUTO_INCREMENT,
   question VARCHAR(255) NOT NULL,
   answer VARCHAR(255) NOT NULL,
   PRIMARY KEY(ID_CAPTCHA)
);

CREATE TABLE TRANSLATIONS(
   id INT AUTO_INCREMENT PRIMARY KEY,
   original_text TEXT NOT NULL,
   translated_text TEXT NOT NULL,
   target_language VARCHAR(5) NOT NULL
);

CREATE TABLE CONTACTS(
   id INT AUTO_INCREMENT PRIMARY KEY,
   subject VARCHAR(100),
   email VARCHAR(80),
   message VARCHAR(200)
);

CREATE TABLE USER_INTERACTION_ADVICE (
    ID_USER INT,
    ID_ADVICE INT,
    PRIMARY KEY (ID_USER, ID_ADVICE),
    FOREIGN KEY (ID_USER) REFERENCES USER_(ID_USER),
    FOREIGN KEY (ID_ADVICE) REFERENCES ADVICE(ID_ADVICE)
);

CREATE TABLE SHOP_ORDER ( 
    ID_SHOP_ORDER INTEGER PRIMARY KEY AUTO_INCREMENT,
    order_date DATE,
    total_price DECIMAL(8,2),
    vat DECIMAL(3,1),
    status INT,
    transaction_id VARCHAR(255) DEFAULT NULL,
    ID_CONSUMER INT,
    FOREIGN KEY(ID_CONSUMER) REFERENCES CONSUMER(ID_CONSUMER)
);

CREATE TABLE ORDER_LINE(
   ID_ORDER_LINE INTEGER PRIMARY KEY AUTO_INCREMENT,
   ID_SHOP_ORDER INT NOT NULL,
   ID_PRODUCT INT NOT NULL,
   quantity INT,
   unit_price DECIMAL(8,2),
   FOREIGN KEY(ID_SHOP_ORDER) REFERENCES SHOP_ORDER(ID_SHOP_ORDER),
   FOREIGN KEY(ID_PRODUCT) REFERENCES PRODUCT(ID_PRODUCT)
);

CREATE TABLE SERVICE_PROVIDER_DOCUMENT(
    ID_DOCUMENT INTEGER PRIMARY KEY AUTO_INCREMENT,
    ID_SERVICE_PROVIDER INT,
    type VARCHAR(100),
    docPath VARCHAR(255),
    uploadDate DATE,
    status BOOLEAN DEFAULT 0,
